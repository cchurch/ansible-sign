NeoPixel

  • Individually Controllable
  • 24-bit RGB
  • 60mA @ 5V (300mW)

NeoPixel Library

#include "Adafruit_NeoPixel.h"

Adafruit_NeoPixel pixels = Adafruit_NeoPixel(97, 12,
  NEO_GRB + NEO_KHZ800);

void setup() {
    pixels.begin();
    pixels.setBrightness(255);
    pixels.show();
}

void loop() {
    for (int i = 0; i < 97; i++) {
        pixels.setPixelColor(i, pixels.Color(255, 255, 255));
    }
    pixels.show();
}
